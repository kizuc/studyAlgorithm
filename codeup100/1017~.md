# 1017. 정수 1개 입력받아 3번 출력
```java
import java.util.Scanner;

public class Main {

	public static void main(String[] args) {
		
		Scanner sc = new Scanner(System.in);
		
		// 1
		int a = sc.nextInt();
		System.out.printf("%d %d %d", a, a, a);
		
		// 2
		int a2 = sc.nextInt();
		for(int i = 0; i < 3; i++) {
			System.out.printf("%d ", a2);
		}

	}
}
```
